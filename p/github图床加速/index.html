addEventListener(
  "fetch",event => {
     let url=new URL(event.request.url);
     url.hostname="raw.githubusercontent.com";  //反代github域名
     let request=new Request(url,event.request);
     event. respondWith(
       fetch(request)
     )
  }
)

具体教程网上很多，需要你在Cloudflare有域名
